        float
            F2(x,Q^2)

        # leading order is just a delta function
        return lambda z: 0, lambda z: 1



[docs]    def quark_1(self):
        """
        regular
        delta
        1/(1-x)_+
        log(x)/(1-x)_+

        .. todo::
            docs
        """
        CF = self._SF._constants.CF
        zeta_2 = np.pi ** 2 / 6

        def cq_reg(z):
            # fmt: off
            return CF*(
                - 2 * (1 + z) * np.log((1 - z) / z)
                - 4 * np.log(z) / (1 - z)
                + 6 + 4 * z
            )
            # fmt: on

        def cq_delta(z):
            return -CF * (9 + 4 * zeta_2)

        def cq_omx(z):
            return -3 * CF

        def cq_logomx(z):
            return 4 * CF

        return cq_reg, cq_delta, cq_omx, cq_logomx

        """
        vogt page 21

        .. todo::

            - 2 * n_f here and in gluon_1_fact is coming from momentum sum rule
            q_i -> {q_i, g} but g -> {g, q_i, \bar{q_i} \forall i}, so the 2*n_f
            is needed to compensate for all the number of flavours + antiflavours
            in which the gluon can go
        """

        TR = self._SF._constants.TF

        def cg(z):
            return (
                2  # TODO: to be understood
                * 2
                * self._n_f
                * (
                    split.pqg(z, self._SF._constants) * (np.log((1 - z) / z) - 4)
                    + 3 * TR
                )
            )

        return cg
